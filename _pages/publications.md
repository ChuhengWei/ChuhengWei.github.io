function showCite(id) {
  var content = document.getElementById(id).textContent;
  document.getElementById('cite-modal-content').textContent = content;
  document.getElementById('cite-modal').style.display = 'flex';
  document.getElementById('copy-btn').textContent = '复制';
}
function closeCite() {
  document.getElementById('cite-modal').style.display = 'none';
}
function copyCite() {
  var text = document.getElementById('cite-modal-content').textContent;
  navigator.clipboard.writeText(text).then(function() {
    document.getElementById('copy-btn').textContent = '已复制 ✓';
  });
}
document.getElementById('cite-modal').addEventListener('click', function(e) {
  if (e.target === this) closeCite();
});
</script>
